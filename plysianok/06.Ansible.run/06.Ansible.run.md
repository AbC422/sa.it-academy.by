## Playbook for the Ansible user (Create new user (use module), new user authorization only by SSH key, add its to the sudo:nopasswd for upgrade command)
---
```bash
- hosts: work_hosts
  vars:
    user: ansi
  tasks:
    - name: "Create new user <ansi> {{ user }}"
      user:
        name: "{{ user }}"
    - name: "Add authorized ssh-keys"
      authorized_key:
        user: "{{ user }}"
        key: "{{ lookup('file', '~/.ssh/id_rsa.pub') }}"
    - name: "Sudo without a password in CentOS"
      when: ansible_distribution == 'CentOS'
      lineinfile:
        dest: "/etc/sudoers"
        state: "present"
        regexp: "{{ user }}"
        line: "{{ user }}  ALL=(ALL) NOPASSWD: /usr/bin/yum"
    - name: "Sudo without a password in Ubuntu"
      when: ansible_distribution == 'Ubuntu'
      lineinfile:
        dest: "/etc/sudoers"
        state: "present"
        regexp: "{{ user }}"
        line: "{{ user }}  ALL=(ALL) NOPASSWD: /usr/bin/apt"
    - name: "User ansi"
      command: |
        sudo grep "{{ user }}" /etc/passwd /etc/sudoers
      register: out
    - debug:
        msg: "{{ out.stdout_lines }}"
    - name: "Upgrade Ubuntu"
      when: ansible_distribution == 'Ubuntu'
      command: |
        sudo apt upgrade -y
      register: ubuntu
    - debug:
        msg: "{{ ubuntu }}"
    - name: "Upgrade CentOS"
      remote_user: "{{ user }}"
      when: ansible_distribution == 'CentOS'
      command: |
        sudo yum upgrade -y
      register: centos
    - debug:
        msg: "{{ centos }}"
```
